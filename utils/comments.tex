\newcounter{mycommentJW}
\newcommandx{\juntang}[2][1=]{
  % initials of the author (optional) + note in the margin
  \refstepcounter{mycommentJW}%
  {%
    \ifthenelse{\equal{#1}{inline}}%
    {%
      \addtocounter{@todonotes@numberoftodonotes}{1}%
      \addcontentsline{tdo}{todo}{%
        \fcolorbox{black}%
        {brown!20!white}%
        {\phantom{o}}%
        \ \textbf{[\uppercase{JW}\themycommentJW]:}~#2}%
      \textcolor{brown}{\textbf{[\uppercase{JW}\themycommentJW]:}~#2}
    }%
    {%
      \todo[color={brown!20!white},#1]{%
        \textbf{[\uppercase{JW}\themycommentJW]:}~#2}%
    }
  }
}
